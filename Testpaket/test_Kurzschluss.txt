// Diese Schaltnetzdatei dient zum Überprüfen,
// ob das Programm Kurzschlüsse erkennt. 

ARCHITECTURE kurzschluss OF Informationstechnik IS

ENTITY
INPUT s001,s002;
OUTPUT s005;
SIGNALS s003,s004;
CLOCK clk, 1500 MHz;

BEGIN
g001:inv1a(s001,s003);
g002:dff(s001,clk,s003);
g003:xor2(s003,s002,s004);
g004:inv1a(s004,s005);
END
